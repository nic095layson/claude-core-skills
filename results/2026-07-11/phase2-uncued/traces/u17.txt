#!/usr/bin/env bash
#
# healthcheck.sh — probe the Acme Orders API liveness endpoint.
# Exits 0 if the service responds with HTTP 2xx, non-zero otherwise.
#
# Note: README.md claims port 8080, but server.js listens on 3000.
# Override with HEALTHCHECK_URL if your deployment differs.

set -euo pipefail

URL="${HEALTHCHECK_URL:-http://localhost:3000/health}"

status="$(curl --silent --show-error --max-time 5 \
  --output /dev/null --write-out '%{http_code}' "$URL")" || {
  echo "healthcheck: no response from $URL" >&2
  exit 1
}

if [ "$status" -ge 200 ] && [ "$status" -lt 300 ]; then
  echo "healthcheck: OK ($status) $URL"
  exit 0
fi

echo "healthcheck: unhealthy (HTTP $status) $URL" >&2
exit 1
