#!/usr/bin/env bash
#
# Health check for the Acme Orders API.
# Curls the service's /health endpoint and exits non-zero if it's not responding.
#
# Usage:
#   ./healthcheck.sh                 # checks http://localhost:$PORT/health
#   PORT=8080 ./healthcheck.sh       # override the port
#   HOST=orders.internal ./healthcheck.sh
#
set -euo pipefail

HOST="${HOST:-localhost}"
PORT="${PORT:-3000}"
URL="http://${HOST}:${PORT}/health"

# --fail makes curl return non-zero on HTTP >= 400; --silent/--show-error keep
# output quiet unless something goes wrong; --max-time bounds a hung server.
if curl --fail --silent --show-error --max-time 5 "${URL}" >/dev/null; then
  echo "OK: ${URL} is healthy"
  exit 0
else
  echo "UNHEALTHY: ${URL} is not responding" >&2
  exit 1
fi

--end--
